/*6. Create a script that selects all elements with the tag name "input"
and logs their type attribute to the console.*/

function getAttributeType(tagname) {
    let allTagsInputs = document.getElementsByTagName(tagname);
    for (let index = 0; index < allTagsInputs.length; index++) {
        const element = allTagsInputs[index];
        console.log(element.type);
    }
}
